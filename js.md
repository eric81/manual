#### 获取checkbox选中项
##### 普通js方式
```javascript
function getOptionArrays(optionName){
	var checkedOptions = [];
    var r = document.getElementsByName(optionName);  
    for(var i=0; i<r.length; i++){
    	if(r[i].checked){
    		checkedOptions.push(r[i]);
    	}
    }
    
    return checkedOptions;
}
```
